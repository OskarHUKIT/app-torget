# Quick Git Setup Commands
# Copy and paste these commands one by one into PowerShell

# 1. Navigate to project
cd "c:\Users\oskar\Desktop\app torget"

# 2. Initialize git
git init

# 3. Add all files
git add .

# 4. Create initial commit
git commit -m "Initial commit - App Torget marketplace"

# 5. After creating GitHub repo, add remote (replace YOUR_USERNAME)
git remote add origin https://github.com/YOUR_USERNAME/app-torget.git

# 6. Push to GitHub
git branch -M main
git push -u origin main
